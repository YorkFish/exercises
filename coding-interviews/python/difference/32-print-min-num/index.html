    32. 把数组排成最小的数Link

    异 sort-cmpLink

    Python2Link

    例一Link

    nums = [3, 32, 321]
nums = map(str, nums)
print nums

print cmp(1, 2)

nums.sort(cmp=lambda x,y: cmp(x+y, y+x))  # 组合后，字典序小的在前
print nums

    >>>

    ['3', '32', '321']
-1
['321', '32', '3']

    例二Link

    # 升序
def my_cmp(a, b):
    if a < b:
        return -1  # a < b，排序后 a 出现在 b 之前，返回一个小于零的值
    elif a == b:
        return 0
    else:
        return 1

nums = [3, 1, 2]
nums.sort(cmp=my_cmp)
print nums

    >>>

    [1, 2, 3]

    Python3Link

    • Python3 没有 cmp()，且取消了 sort(), sorted() 的 cmp 参数
    from functools import cmp_to_key


# 降序
def my_cmp(a, b):
    if a < b:
        return 1  # a < b，排序后 a 出现在 b 之后，返回一个大于零的值
    elif a == b:
        return 0
    else:
        return -1


nums = [3, 1, 2]
nums.sort(key=cmp_to_key(my_cmp))
print(nums)

    >>>

    [3, 2, 1]
